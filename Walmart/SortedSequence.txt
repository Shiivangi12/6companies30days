vector<int> find3Numbers(vector<int> arr, int N) {
        // Your code here
        int small[N+1] = {0}, large[N+1] = {0};
        int mn = INT_MAX, mx = INT_MIN;
        for(int i=0; i<N-1; i++){
            mn = std::min(mn, arr[i]);
            small[i+1] = mn;
        }
        for(int i=N-1; i>0; i--){
            mx = std::max(mx, arr[i]);
            large[i-1] = mx;
        }
        for(int i=1; i<N-1; i++){
            if(arr[i] > small[i] && arr[i] < large[i])
                return {small[i], arr[i], large[i]};
        }
        return {};
    }
